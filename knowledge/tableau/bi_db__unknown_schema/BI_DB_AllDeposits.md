left join BI_DB.[dbo].[BI_DB_AllDeposits] ad on ad.CID=tasks.CID
and IsFTD=1
left join DWH.dbo.Dim_Customer cc on cc.RealCID=tasks.CID
left join DWH.dbo.Dim_Country dc on dc.CountryID=cc.CountryID
left join DWH.dbo.Dim_Regulation dr on dr.ID=cc.DesignatedRegulationID

count(distinct bd.CID) as FTDs,

 bd.ModificationDate>=dateadd(day,-30,getdate())
